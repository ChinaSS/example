Util

该组件主要放一些代码量较少、常用的公共方法

注意： 该模块依赖dialog.js，且规定在requireJS的config中定义为Dialog

alert方法

  require(["Util/util"],function(util){
      util.alert("提示消息");
  });

confirm方法

  require(["Util/util"],function(util){
      util.confirm("确认提交?",function(){
        //点击确定调用 
 
      },function(){
        //点击取消调用 
 
      });
  });

contentDialog方法

/**
 * config = {
 *   setting : {},                   //dialogSetting
 *   template : templateURL,         //dialogBodyTemplate
 *   afterLoad : function(dialog){},   //afterDialogLoaded callback
 * }
 */
require(["Util/util"],function(util){
   util.contentDialog({
        template : getStaticPath()+"app/views/contentDialog.html",
        afterLoad : function(dialog){
            console.log(dialog);
        }
   })
});

slidebar方法

侧边弹出栏

参数说明

参数名称 是否可选 说明
id 可选 当侧边栏的html已经在页面上时使用,侧边栏html根节点id
url 可选 远程加载侧边栏html模板，从远程拿过来后会自动追加到body上
width 可选 侧边栏宽度
cache 可选 是否需要缓存，默认为true，当通过url远程加载模板时，只加载一次
allowClick 可选 数组，侧边栏之外的区域可以点击不关闭侧边栏的DOM节点

样例

  require(["Util/util"],function(util){
      util.slidebar({
          id:"resourceEditPanel",
          width:"500px",
          afterLoad:function(){
              //绑定侧边栏中字段的表单验证 
              validate();
          }
      });
  });

template方法

简单模板引擎

占位替换样例

  require(["Util/util"],function(util){
      var temp  = "<span><%=title%></span>"
      var model = {title:"hi"}
      //第一个参数可以为id或为字符串，第二个参数为模型 
      var result = util.template(temp,model);
      //输出：<span>hi</span> 
      console.log(result);
  });

for与if...else...综合样例

  <!-- html模板 -->
  <script id="T_breadcrumbs" type="text/html">
      <li>
          <i class="ace-icon fa fa-home home-icon"></i>
          <a href="index.html">首页</a>
      </li>
      <%for(var i=items.length;i--;){%>
          <%if(i==0){%>
              <li class="active"><%=items[i].name %></li>
          <%}else{%>
              <li><a href="#"><%=items[i].name %></a></li>
          <%}%>
      <%}%>
  </script> 
  //js代码 
  require(["Util/util"],function(util){
      //更新导航条 
      var breadcrumb_items = [];
      //循环追加内容 
      $(li.firstChild).parents('.nav-list li').each(function() {
          var link = $(this).find('> a');
          var text = link.text();
          var href = link.attr('href');
          breadcrumb_items.push({'name': text, 'href': href});
      });
      //把model和模板合并后的内容追加到$(".breadcrumb")元素上 
      $(".breadcrumb").html(util.template("T_breadcrumbs",{items:breadcrumb_items}));
  });
